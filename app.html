async function runSearch() { const isAdmin = ADMIN_EMAILS.includes(currentUser.email.toLowerCase()); const hoje = new Date().toISOString().split('T')[0]; // Verificação de limite if (!isAdmin) { const { count } = await sb.from('logs_buscas').select('*', { count: 'exact', head: true }).eq('user_id', currentUser.id).eq('created_at', hoje); if (count >= 25) return showToast("🚫 Limite diário atingido!"); } const btn = document.getElementById('btnSearch'); btn.disabled = true; btn.innerHTML = '
BUSCANDO...'; document.getElementById('scraperResults').innerHTML = ''; const payload = { apiChoice: document.getElementById('apiChoice').value, loc: document.getElementById('srcLoc').value, seg: document.getElementById('srcSeg').value || document.getElementById('srcNicho').value, limit: document.getElementById('srcLimit').value }; try { // CHAMADA PARA O SUPABASE EDGE FUNCTION const { data, error } = await sb.functions.invoke('search-leads', { body: payload }); if (error) throw error; if (data.results && data.results.length > 0) { lastResults = data.results; // Registra no limite se não for admin if (!isAdmin) { for(let i=0; i < data.results.length; i++) { await sb.from('logs_buscas').insert({ user_id: currentUser.id }); } } renderResults(data.results); showToast(`${data.results.length} leads encontrados!`); } else { showToast("Nenhum lead encontrado."); } } catch (e) { console.error(e); showToast("Erro na busca: Tente novamente em instantes."); } finally { btn.disabled = false; btn.innerHTML = 'BUSCAR PROSPECTOS'; } }
